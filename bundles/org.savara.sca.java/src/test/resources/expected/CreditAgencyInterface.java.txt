package org.jboss.examples.creditagency;

import javax.jws.WebMethod;
import javax.jws.WebParam;
import javax.jws.WebResult;
import javax.jws.WebService;
import javax.jws.soap.SOAPBinding;
import javax.xml.bind.annotation.XmlSeeAlso;


@WebService(targetNamespace = "http://www.jboss.org/examples/creditAgency", name = "CreditAgencyInterface")
@XmlSeeAlso({ObjectFactory.class})
@SOAPBinding(parameterStyle = SOAPBinding.ParameterStyle.BARE)
@org.oasisopen.sca.annotation.Remotable public interface CreditAgencyInterface {

    @WebResult(name = "CreditRating", targetNamespace = "http://www.jboss.org/examples/creditAgency", partName = "content")
    @WebMethod(action = "http://www.jboss.org/examples/creditAgency/checkCredit")
    public CreditRating checkCredit(
        @WebParam(partName = "content", name = "CreditCheckRequest", targetNamespace = "http://www.jboss.org/examples/creditAgency")
        CreditCheckRequest content
    ) throws CustomerUnknownFault;
}
